function run(taskDef) {
    let task = taskDef();
    let result = task.next();

    function step() {
        if (!result.done) {
            result = task.next(result.value);
            step();
        }
    }
    step();
}

function fetchData() {
    return function(callback) {
        setTimeout(() => callback(null, "Hi!"), 50);
    };
}

run(function*() {
    let data = yield fetchData();
    console.log(data); // "Hi!"
});
